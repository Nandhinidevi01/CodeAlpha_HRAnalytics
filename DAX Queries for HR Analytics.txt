ðŸ”¹ 2. Turnover Rate

EVALUATE
VAR TotalEmployees = COUNTROWS('HRDataset_v14')
VAR EmployeesLeft = CALCULATE(
    COUNTROWS('HRDataset_v14'),
    NOT(ISBLANK('HRDataset_v14'[DateofTermination]))
)
RETURN
ROW("Turnover Rate (%)", DIVIDE(EmployeesLeft, TotalEmployees, 0) * 100)
